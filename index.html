squares.js

Get straight to the repo here.

A bisecting, treemap-like visualization of tags, all wrapped up in a nice jQuery plugin.

Sane Defaults

<div id="squares">
    <a href="#squares" weight="2">BIPOLAR</a> 
    <a href="#squares" weight="1">ECL</a> 
    <a href="#squares" weight="1">TTL</a> 
    <a href="#squares" weight="80">CMOS</a> 
    <a href="#squares" weight="12">NMOS</a> 
    <a href="#squares" weight="20">HCMOS</a> 
    <a href="#squares" weight="19">MOS</a> 
</div>

$('#squares a').squares();

BIPOLAR ECL TTL CMOS NMOS HCMOS MOS

Optional Parameters

<div id="squares_2"> 
    <a href="#squares_2" weight="7">7</a>
    <a href="#squares_2" weight="4">4</a> 
    <a href="#squares_2" weight="3">3</a> 
    <a href="#squares_2" weight="5">5</a> 
</div>

$('#squares a').squares({
    colors: ["#778899","LightBlue","rgba(81,81,81,1)"], 
    min: 25, 
    max: 40
});
7 4 3 5
